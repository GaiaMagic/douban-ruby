  def get_book(id=nil)
    if id.to_s.size >=10
      resp=get("/book/subject/isbn/#{url_encode(id.to_s)}")
    else
      resp=get("/book/subject/#{url_encode(id.to_s)}")
    end
    if resp.code=="200"
      atom=resp.body
      Book.new(atom)
    else
      nil
    end
  end
